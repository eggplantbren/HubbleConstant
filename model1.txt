# A simple model where the quoted error bars
# (sig) may be wrong, by replacing the normal
# distribution with a t-distribution.

model
{
	# Cauchy prior on H0
	H0 ~ dt(0., pow(100., -2), 1.)

	# Heavy-tailedness of likelihood
	log_nu ~ dunif(0., 5.)

	# Data
	for(i in 1:N)
	{
		y[i] ~ dt(H0, pow(sig[i], -2), exp(log_nu))
	}
}
